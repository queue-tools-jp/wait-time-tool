function calculate(){ const pts = collect(); const power = Number(document.getElementById('weightPower').value) || 2; if(pts.length < 2){ document.getElementById('result').innerText = '2点以上入力してください。'; return; } // -------------------------------- // STEP1: 区間速度収集 // -------------------------------- let rawSegments = []; let speeds = []; for(let i=0;i 0 && dq > 0){ const speed = dq / dt; rawSegments.push({ i, dt, dq, speed }); speeds.push(speed); } } if(speeds.length === 0){ document.getElementById('result').innerText = '有効な進行データがありません。'; return; } // -------------------------------- // STEP2: 中央値 // -------------------------------- const sorted = [...speeds].sort((a,b)=>a-b); const median = sorted[Math.floor(sorted.length / 2)]; // 異常減少判定 const abnormalThreshold = median * 5; // 加速上限制限 const speedCap = median * 1.5; // -------------------------------- // STEP3: 重み付き平均 // -------------------------------- let wt = 0; let wp = 0; let segments = []; const n = rawSegments.length; rawSegments.forEach((seg, idx)=>{ // 異常減少 const abnormal = seg.speed > abnormalThreshold; seg.abnormal = abnormal; if(abnormal) return; // -------------------------- // 加速制限 // -------------------------- let adjustedSpeed = seg.speed; if(seg.speed > speedCap){ adjustedSpeed = speedCap; } // 補正後の分/人 const adjustedMpp = 1 / adjustedSpeed; const adjustedDt = adjustedMpp * seg.dq; // -------------------------- // 重み // -------------------------- const w = Math.pow((idx+1)/n, power); wt += adjustedDt * w; wp += seg.dq * w; segments.push({ ...seg, adjustedSpeed }); }); if(wp <= 0){ document.getElementById('result').innerText = '正常データが不足しています。'; return; } // -------------------------------- // STEP4: 平均 // -------------------------------- const mpp = wt / wp; const perHour = 60 / mpp; // -------------------------------- // STEP5: 各行表示 // -------------------------------- for(let i=0;i<=MAX;i++){ const e = document.getElementById(`ref${i}`); if(e) e.innerHTML=''; } pts.forEach((p, idx)=>{ const arr = fmt(toMin(p.t) + p.q * mpp); let trust = '高'; const seg = segments.find(s=>s.i===idx); const rawSeg = rawSegments.find(s=>s.i===idx); if(rawSeg?.abnormal){ trust = '異常除外'; } else if(seg){ // 加速補正表示 if(seg.speed > speedCap){ trust = '加速補正'; } else { const local = seg.dt / seg.dq; const diff = Math.abs(local - mpp); if(diff > mpp * 0.5){ trust = '低'; } else if(diff > mpp * 0.2){ trust = '中'; } } } const target = document.getElementById(`ref${p.index}`); if(target){ target.innerHTML = `予測:${arr}
信頼:${trust}`; } }); // -------------------------------- // STEP6: 最終予測 // -------------------------------- const last = pts[pts.length-1]; const remainMin = last.q * mpp; const arrival = fmt(toMin(last.t) + remainMin); // 所要時間 const base = pts[0]; const baseMin = toMin(base.t); const arrivalMin = toMin(last.t) + remainMin; const totalMin = diffMin(baseMin, arrivalMin); // -------------------------------- // STEP7: 異常一覧 // -------------------------------- const abnormalList = rawSegments .filter(s=>s.abnormal) .map(s=>`#${s.i}→#${s.i+1}`) .join(', '); // -------------------------------- // STEP8: 結果表示 // -------------------------------- document.getElementById('result').innerHTML = ` 中央値速度： ${median.toFixed(2)} 人 / 分
加速上限： ${speedCap.toFixed(2)} 人 / 分
異常判定： ${abnormalThreshold.toFixed(2)} 人 / 分 超
重み付き平均： ${mpp.toFixed(2)} 分 / 人
1時間あたり： 約 ${Math.round(perHour)} 人 / 時
残り時間： 約 ${Math.round(remainMin)} 分
所要時間（目安）： ${fmtHM(totalMin)}
到達予測： ${arrival}
異常除外区間： ${abnormalList || 'なし'} `; }
